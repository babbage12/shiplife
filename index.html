        function createGlobe() {

            const textureLoader = new THREE.TextureLoader();

            

            // NASA Blue Marble 8K texture hosted on Imgur

            const earthTexture = textureLoader.load(

                'https://i.imgur.com/TIrHFCa.jpg',

                function() { 

                    console.log('Earth texture loaded');

                }

            );

            

            // Bump map for terrain elevation

            const bumpMap = textureLoader.load(

                'https://unpkg.com/three-globe/example/img/earth-topology.png'

            );

            

            // Specular map - makes oceans shiny, land matte

            const specularMap = textureLoader.load(

                'https://unpkg.com/three-globe/example/img/earth-water.png'

            );


            // Higher polygon count for smoother sphere

            const geometry = new THREE.SphereGeometry(1, 128, 128);

            

            // Enhanced material with specular highlights on water

            const material = new THREE.MeshPhongMaterial({

                map: earthTexture,

                bumpMap: bumpMap,

                bumpScale: 0.015,

                specularMap: specularMap,

                specular: new THREE.Color(0x666666),

                shininess: 25

            });


            globe = new THREE.Mesh(geometry, material);

            scene.add(globe);


            // Enhanced atmosphere glow - multiple layers for depth

            // Inner atmosphere (subtle blue rim)

            const innerAtmoGeometry = new THREE.SphereGeometry(1.015, 64, 64);

            const innerAtmoMaterial = new THREE.ShaderMaterial({

                vertexShader: `

                    varying vec3 vNormal;

                    void main() {

                        vNormal = normalize(normalMatrix * normal);

                        gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);

                    }

                `,

                fragmentShader: `

                    varying vec3 vNormal;

                    void main() {

                        float intensity = pow(0.75 - dot(vNormal, vec3(0.0, 0.0, 1.0)), 2.5);

                        vec3 atmosphereColor = vec3(0.4, 0.7, 1.0);

                        gl_FragColor = vec4(atmosphereColor, intensity * 0.8);

                    }

                `,

                blending: THREE.AdditiveBlending,

                side: THREE.BackSide,

                transparent: true,

                depthWrite: false

            });

            const innerAtmosphere = new THREE.Mesh(innerAtmoGeometry, innerAtmoMaterial);

            globe.add(innerAtmosphere);

            

            // Outer atmosphere (soft glow extending further)

            const outerAtmoGeometry = new THREE.SphereGeometry(1.12, 64, 64);

            const outerAtmoMaterial = new THREE.ShaderMaterial({

                vertexShader: `

                    varying vec3 vNormal;

                    void main() {

                        vNormal = normalize(normalMatrix * normal);

                        gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);

                    }

                `,

                fragmentShader: `

                    varying vec3 vNormal;

                    void main() {

                        float intensity = pow(0.65 - dot(vNormal, vec3(0.0, 0.0, 1.0)), 3.0);

                        vec3 glowColor = vec3(0.3, 0.5, 0.9);

                        gl_FragColor = vec4(glowColor, intensity * 0.4);

                    }

                `,

                blending: THREE.AdditiveBlending,

                side: THREE.BackSide,

                transparent: true,

                depthWrite: false

            });

            const outerAtmosphere = new THREE.Mesh(outerAtmoGeometry, outerAtmoMaterial);

            globe.add(outerAtmosphere);

        }


        function latLonToVector3(lat, lon, radius) {

            const phi = (90 - lat) * (Math.PI / 180);

            const theta = (lon + 180) * (Math.PI / 180);

            

            return new THREE.Vector3(

                -radius * Math.sin(phi) * Math.cos(theta),

                radius * Math.cos(phi),

                radius * Math.sin(phi) * Math.sin(theta)

            );

        }

            // Helper to draw a lush palm tree

            function drawPalm(x, baseY, trunkH, lean, size) {

                ctx.fillStyle = '#0a0a0a';

                

                // Curved trunk

                ctx.beginPath();

                ctx.moveTo(x - 4 * size, baseY);

                ctx.quadraticCurveTo(x + lean * 0.3, baseY - trunkH * 0.3, x + lean * 0.6, baseY - trunkH * 0.6);

                ctx.quadraticCurveTo(x + lean * 0.8, baseY - trunkH * 0.8, x + lean, baseY - trunkH);

                ctx.lineTo(x + lean + 4 * size, baseY - trunkH);

                ctx.quadraticCurveTo(x + lean * 0.8 + 4 * size, baseY - trunkH * 0.8, x + lean * 0.6 + 4 * size, baseY - trunkH * 0.6);

                ctx.quadraticCurveTo(x + lean * 0.3 + 4 * size, baseY - trunkH * 0.3, x + 4 * size, baseY);

                ctx.closePath();

                ctx.fill();

                

                // Fronds - big drooping leaves

                const cx = x + lean + 2 * size;

                const cy = baseY - trunkH;

                

                for (let i = 0; i < 9; i++) {

                    const angle = (i / 9) * Math.PI * 1.6 - Math.PI * 0.8;

                    const len = (50 + Math.random() * 30) * size;

                    const droop = 0.4 + Math.random() * 0.2;

                    

                    // Main frond stem

                    ctx.beginPath();

                    ctx.moveTo(cx, cy);

                    

                    const midX = cx + Math.cos(angle) * len * 0.5;

                    const midY = cy + Math.sin(angle) * len * 0.3;

                    const endX = cx + Math.cos(angle) * len;

                    const endY = cy + Math.sin(angle) * len * droop + len * 0.2;

                    

                    ctx.quadraticCurveTo(midX, midY, endX, endY);

                    ctx.quadraticCurveTo(midX + 2, midY + 2, cx, cy);

                    ctx.fill();

                    

                    // Leaflets along the frond

                    for (let j = 0; j < 6; j++) {

                        const t = 0.2 + j * 0.13;

                        const lx = cx + (midX - cx) * t * 2;

                        const ly = cy + (endY - cy) * t;

                        const leafLen = 15 * size * (1 - t * 0.5);

                        const leafAngle = angle + (j % 2 === 0 ? 0.4 : -0.4);

                        

                        ctx.beginPath();

                        ctx.moveTo(lx, ly);

                        ctx.lineTo(lx + Math.cos(leafAngle) * leafLen, ly + Math.sin(leafAngle) * leafLen * 0.5);

                        ctx.lineTo(lx + 2, ly + 2);

                        ctx.fill();

                    }

                }

                

                // Coconuts

                ctx.beginPath();

                ctx.arc(cx - 3 * size, cy + 5 * size, 4 * size, 0, Math.PI * 2);

                ctx.arc(cx + 3 * size, cy + 7 * size, 3.5 * size, 0, Math.PI * 2);

                ctx.arc(cx, cy + 9 * size, 3 * size, 0, Math.PI * 2);

                ctx.fill();

            }

            

            // Helper to draw tropical ferns/bushes

            function drawFern(x, baseY, size) {

                ctx.fillStyle = '#0a0a0a';

                for (let i = 0; i < 7; i++) {

                    const angle = (i / 7) * Math.PI - Math.PI * 0.5;

                    const len = (30 + Math.random() * 20) * size;

                    

                    ctx.beginPath();

        function renderDoor1Frame() {

            if (!door1Renderer) return;

            

            const time = Date.now();

            const doorBaseX = -0.7;

            

            // Animation phases

            if (door1AnimPhase === 'shake') {

                const elapsed = time - door1AnimStart;

                if (elapsed < 600) {

                    const intensity = Math.sin(elapsed * 0.06) * 0.015 * (1 - elapsed / 600);

                    door1Group.rotation.z = intensity;

                    door1Group.position.x = doorBaseX + Math.sin(elapsed * 0.1) * 0.008;

                } else {

                    door1Group.rotation.z = 0;

                    door1Group.position.x = doorBaseX;

                    door1AnimPhase = 'unlock';

                    door1AnimStart = time;

                }

            } else if (door1AnimPhase === 'unlock') {

                const elapsed = time - door1AnimStart;

                if (elapsed < 400) {

                    const progress = elapsed / 400;

                    const easeOut = 1 - Math.pow(1 - progress, 3);

                    door1ShackleGroup.rotation.z = easeOut * 0.7;

                    door1ShackleGroup.position.y = easeOut * 0.08;

                } else {

                    door1AnimPhase = 'fall';

                    door1AnimStart = time;

                    door1IsLocked = false;

                }

            } else if (door1AnimPhase === 'fall') {

                const elapsed = time - door1AnimStart;

                if (elapsed < 300) {

                    const progress = elapsed / 300;

                    door1LockGroup.position.y = -0.15 - progress * 1.5;

                    door1LockGroup.rotation.x = progress * 0.8;

                    door1LockGroup.rotation.z = progress * 0.4;

                } else {

                    door1LockGroup.visible = false;

                    door1AnimPhase = 'swing';

                    door1AnimStart = time;

                }

            } else if (door1AnimPhase === 'swing') {

                const elapsed = time - door1AnimStart;

                if (elapsed < 800) {

                    const progress = elapsed / 800;

                    const easeOut = 1 - Math.pow(1 - progress, 3);

                    const bounce = Math.sin(progress * Math.PI * 2) * 0.04 * (1 - progress);

                    door1Group.rotation.y = (easeOut + bounce) * -1.2;

                    

                    // Show sky backdrop once door is partially open (20%)

                    if (progress > 0.2 && window.skyPlane) {

                        window.skyPlane.visible = true;

                    }

                    

                    // Start jungle animation when door is 40% open

                    if (progress > 0.4 && window.jungleAnimStart === 0) {

                        window.jungleAnimStart = time;

                    }

                } else {

                    door1Group.rotation.y = -1.2;

                    door1AnimPhase = 'open';

                    if (window.jungleAnimStart === 0) {

                        window.jungleAnimStart = time;

                    }

                }

            }

            

            // Animate trees popping forward (runs during swing and open phases)

            if (window.jungleAnimStart > 0 && window.palmTrees) {

                const jungleElapsed = time - window.jungleAnimStart;

                

                window.palmTrees.forEach((tree, i) => {

                    const treeElapsed = jungleElapsed - tree.userData.delay;

                    

                    if (treeElapsed > 0) {

                        tree.visible = true;

                        const popDuration = 800; // Each tree takes 800ms to pop out

                        const progress = Math.min(treeElapsed / popDuration, 1);

                        

                        // Elastic/bouncy easing for lazy pop effect

                        let ease;

                        if (progress < 1) {

                            const t = progress;

                            ease = 1 - Math.pow(1 - t, 3) * Math.cos(t * Math.PI * 1.5) * (1 - t);

                            ease = Math.min(1.1, ease);

                        } else {

                            ease = 1;

                        }

                        

                        // Animate Z position (pop forward out of the door)

                        if (tree.userData.baseZ !== undefined) {

                            tree.position.z = tree.userData.baseZ + (tree.userData.targetZ - tree.userData.baseZ) * Math.min(ease, 1);

                        }

                        

                        // Animate Y position (pop up slightly)

                        tree.position.y = tree.userData.baseY + (tree.userData.targetY - tree.userData.baseY) * Math.min(ease, 1);

                        

                        // Animate scale (grow from tiny)

                        const scaleEase = 1 - Math.pow(1 - Math.min(progress, 1), 2);

                        tree.scale.setScalar(tree.userData.targetScale * scaleEase);

                        

                        // Gentle sway once popped

                        if (progress >= 1) {

                            tree.rotation.z = Math.sin(time * 0.0015 + i * 1.5) * 0.03;

                        }

                    }

                });

                

                // Animate birds after trees

                if (window.birds && window.birds.length > 0) {

                    window.birds.forEach((bird, i) => {

                        const birdElapsed = jungleElapsed - bird.userData.delay;

                        

                        if (birdElapsed > 0) {

                            bird.visible = true;

                            const t = birdElapsed * 0.001 * bird.userData.speed;

                            

                            // Fly across the scene

                            bird.position.x = bird.userData.startX + Math.sin(t) * 0.4;

                            bird.position.y = bird.userData.startY + Math.sin(t * 1.3) * 0.15;

                            

                            // Wing flap

                            bird.children.forEach((wing, wi) => {

                                wing.rotation.z = (wi === 0 ? 1 : -1) * (0.3 + Math.sin(t * 15) * 0.25);

                            });

                        }

                    });

                }

            }

            

            // Idle glow pulse

            if (door1IsLocked && door1AnimPhase === 'idle') {

                const pulse = Math.sin(time * 0.004) * 0.15 + 0.85;

                door1GoldMat.emissiveIntensity = 0.25 * pulse;

            }

            

            door1Renderer.render(door1Scene, door1Camera);

            

            // Update sprite texture

            if (door1Texture) {

                door1Texture.needsUpdate = true;

            }

        }

        function renderDoor2Frame() {

            if (!door2Renderer) return;

            

            const time = Date.now();

            const doorBaseX = -0.7;

            

            // Animation phases for watertight door

            if (door2AnimPhase === 'knock') {

                // Phase 1: Lever turns (unlocking mechanism)

                const elapsed = time - door2AnimStart;

                if (elapsed < 600) {

                    const progress = elapsed / 600;

                    const easeOut = 1 - Math.pow(1 - progress, 3);

                    // Lever rotates 180 degrees

                    if (door2LeverGroup) {

                        door2LeverGroup.rotation.z = easeOut * Math.PI;

                    }

                } else {

                    door2AnimPhase = 'turn';

                    door2AnimStart = time;

                }

            } else if (door2AnimPhase === 'turn') {

                // Phase 2: Bar slides/retracts

                const elapsed = time - door2AnimStart;

                if (elapsed < 500) {

                    const progress = elapsed / 500;

                    const easeOut = 1 - Math.pow(1 - progress, 3);

                    // Bar slides to the side (retracting)

                    if (door2BarGroup) {

                        door2BarGroup.position.x = 1.0 + easeOut * 0.4;

                        door2BarGroup.scale.x = 1 - easeOut * 0.3;

                    }

                } else {

                    door2AnimPhase = 'swing';

                    door2AnimStart = time;

                }

            } else if (door2AnimPhase === 'swing') {

                const elapsed = time - door2AnimStart;

                if (elapsed < 1000) {

                    const progress = elapsed / 1000;

                    const easeOut = 1 - Math.pow(1 - progress, 3);

                    const bounce = Math.sin(progress * Math.PI * 2) * 0.03 * (1 - progress);

                    // Heavy door swings open slowly

                    door2Group.rotation.y = (easeOut + bounce) * -1.2;

                    

                    // Show stage backdrop

                    if (progress > 0.2 && window.door2StagePlane) {

                        if (window.door2Backdrop) window.door2Backdrop.visible = true;

                        window.door2StagePlane.visible = true;

                        if (window.door2Vignette) window.door2Vignette.visible = true;

                    }

                    

                    // Start music animation

                    if (progress > 0.3 && window.door2MusicAnimStart === 0) {

                        window.door2MusicAnimStart = time;

                    }

                } else {

                    door2Group.rotation.y = -1.2;

                    door2AnimPhase = 'open';

                    door2IsOpen = true;

                }

            }

            

            // Animate musical notes floating out

            if (window.door2MusicAnimStart > 0 && window.musicalNotes) {

                const musicElapsed = time - window.door2MusicAnimStart;

                

                window.musicalNotes.forEach((note, i) => {

                    const noteElapsed = musicElapsed - note.userData.delay;

                    

                    if (noteElapsed > 0) {

                        note.visible = true;

                        const popDuration = 600;

                        const progress = Math.min(noteElapsed / popDuration, 1);

                        

                        // Bouncy pop out

                        let ease;

                        if (progress < 1) {

                            const t = progress;

                            ease = 1 - Math.pow(1 - t, 3) * Math.cos(t * Math.PI * 1.5) * (1 - t);

                            ease = Math.min(1.1, ease);

                        } else {

                            ease = 1;

                        }

                        

                        // Pop forward and up

                        note.position.z = note.userData.baseZ + (note.userData.targetZ - note.userData.baseZ) * Math.min(ease, 1);

                        note.position.y = note.userData.baseY + (note.userData.targetY - note.userData.baseY) * Math.min(ease, 1);

                        

                        // Scale up

                        const scaleEase = 1 - Math.pow(1 - Math.min(progress, 1), 2);

                        note.scale.setScalar(note.userData.targetScale * scaleEase);

                        

                        // Continuous floating and rotating once popped

                        if (progress >= 1) {

                            const floatTime = (noteElapsed - popDuration) * 0.001;

                            note.position.y = note.userData.targetY + Math.sin(floatTime * note.userData.floatSpeed + note.userData.floatOffset) * 0.1;

                            note.position.x += Math.sin(floatTime * 0.3 + i) * 0.001;

                            note.rotation.z = Math.sin(floatTime * 0.5 + note.userData.floatOffset) * 0.2;

                        }

                    }

                });

            }

            

            // Idle brass glow pulse

            if (!door2IsOpen && door2AnimPhase === 'idle') {

                const pulse = Math.sin(time * 0.003) * 0.15 + 0.85;

                door2MetalMat.emissiveIntensity = 0.2 * pulse;

            }

            

            door2Renderer.render(door2Scene, door2Camera);

            

            if (door2Texture) {

                door2Texture.needsUpdate = true;

            }

        }

        function focusLocation(loc) {

            const lon = loc.coords[1]; // longitude (negative = west)

            const lat = loc.coords[0]; // latitude (positive = north)

            

            // CALIBRATED VALUES (sorted by longitude):

            // Toledo:       lon=-83.54,  Y=-0.10

            // Búzios:       lon=-41.89,  Y=-0.87

            // Marseille:    lon=5.37,    Y=-1.65

            // Portofino:    lon=9.21,    Y=4.60

            // Rome:         lon=12.50,   Y=4.50

            // Malta:        lon=14.51,   Y=4.42

            // Darwin:       lon=130.85,  Y=2.44

            // Sydney:       lon=151.21,  Y=2.10

            // Brisbane:     lon=153.03,  Y=8.30

            // Milford:      lon=167.90,  Y=8.10

            // Auckland:     lon=174.78,  Y=8.01

            // Tauranga:     lon=176.17,  Y=7.93

            // Napier:       lon=176.91,  Y=7.90

            

            let targetY;

            

            if (lon <= -83.54) {

                // West of Toledo - extrapolate

                const slope = (-0.87 - (-0.10)) / (-41.89 - (-83.54));

                targetY = -0.10 + (lon - (-83.54)) * slope;

            } else if (lon <= -41.89) {

                // Toledo to Búzios

                const t = (lon - (-83.54)) / (-41.89 - (-83.54));

                targetY = -0.10 + t * (-0.87 - (-0.10));

            } else if (lon <= 5.37) {

                // Búzios to Marseille

                const t = (lon - (-41.89)) / (5.37 - (-41.89));

                targetY = -0.87 + t * (-1.65 - (-0.87));

            } else if (lon <= 9.21) {

                // Marseille to Portofino (wrap happens here)

                const t = (lon - 5.37) / (9.21 - 5.37);

                targetY = -1.65 + t * (4.60 - (-1.65));

            } else if (lon <= 12.50) {

                // Portofino to Rome

            } else if (lon <= 130.85) {

                // Malta to Darwin

                const t = (lon - 14.51) / (130.85 - 14.51);

                targetY = 4.42 + t * (2.44 - 4.42);

            } else if (lon <= 151.21) {

                // Darwin to Sydney

                const t = (lon - 130.85) / (151.21 - 130.85);

                targetY = 2.44 + t * (2.10 - 2.44);

            } else if (lon <= 153.03) {

                // Sydney to Brisbane (wrap happens here)

                const t = (lon - 151.21) / (153.03 - 151.21);

                targetY = 2.10 + t * (8.30 - 2.10);

            } else if (lon <= 167.90) {

                // Brisbane to Milford Sound

        // Touch events

        let touchStartX, touchStartY;

        let initialPinchDistance = null;

        let initialCameraZ = null;

        

        function onTouchStart(event) {

            // Hide tooltip when user starts interacting

            hideTooltip();

            

            if (event.touches.length === 1) {

                isUserInteracting = true;

                touchStartX = event.touches[0].clientX;

                touchStartY = event.touches[0].clientY;

                targetRotationXOnMouseDown = targetRotationX;

                targetRotationYOnMouseDown = targetRotationY;

            } else if (event.touches.length === 2) {

                // Pinch zoom start

                isUserInteracting = true;

                initialPinchDistance = getPinchDistance(event.touches);

                initialCameraZ = camera.position.z;

            }

        }


        function onTouchMove(event) {

            event.preventDefault();

            

            if (event.touches.length === 1 && isUserInteracting && !initialPinchDistance) {

                // Single finger drag to rotate

                targetRotationY = targetRotationYOnMouseDown + (event.touches[0].clientX - touchStartX) * 0.002;

                targetRotationX = targetRotationXOnMouseDown + (event.touches[0].clientY - touchStartY) * 0.002;

                targetRotationX = Math.max(-Math.PI / 2, Math.min(Math.PI / 2, targetRotationX));

            } else if (event.touches.length === 2 && initialPinchDistance) {

                // Pinch to zoom

                const currentDistance = getPinchDistance(event.touches);

                const scale = initialPinchDistance / currentDistance;

                camera.position.z = Math.max(1.2, Math.min(4, initialCameraZ * scale));

            }

        }


        function onTouchEnd(event) {

            if (event.touches.length === 0) {

                isUserInteracting = false;

                initialPinchDistance = null;

                initialCameraZ = null;

            } else if (event.touches.length === 1) {

                // Switched from pinch to single touch

                initialPinchDistance = null;

                touchStartX = event.touches[0].clientX;

                touchStartY = event.touches[0].clientY;

                targetRotationXOnMouseDown = targetRotationX;

                targetRotationYOnMouseDown = targetRotationY;

            }

        }

        

        function getPinchDistance(touches) {

            const dx = touches[0].clientX - touches[1].clientX;

            const dy = touches[0].clientY - touches[1].clientY;

            return Math.sqrt(dx * dx + dy * dy);

        }

        function checkHover(event) {

            mouse.x = (event.clientX / window.innerWidth) * 2 - 1;

            mouse.y = -(event.clientY / window.innerHeight) * 2 + 1;

            

            raycaster.setFromCamera(mouse, camera);

            const intersects = raycaster.intersectObjects(markers);

            

            if (intersects.length > 0) {

                const marker = intersects[0].object;

                

                if (marker.userData.title) {

                    hoveredMarker = marker;

                    showTooltip(marker.userData, event.clientX, event.clientY);

                    document.body.style.cursor = 'pointer';

                    

                    // Highlight marker - scale up from original size

                    const baseScale = marker.userData.baseSize * 2;

                    marker.scale.set(baseScale * 1.3, baseScale * 1.3, 1);

                }

            } else {

                if (hoveredMarker) {

                    // Reset to original size

                    const baseScale = hoveredMarker.userData.baseSize * 2;

                    hoveredMarker.scale.set(baseScale, baseScale, 1);

                }

                hoveredMarker = null;

                hideTooltip();

                document.body.style.cursor = isUserInteracting ? 'grabbing' : 'grab';

            }

        }

        function animate() {

            requestAnimationFrame(animate);

            

            // Cinematic intro - just zoom, globe is already positioned

            if (!introComplete) {

                if (!introStartTime) {

                    introStartTime = Date.now();

                    // Set globe rotation immediately to Toledo

                    globe.rotation.y = TOLEDO_Y;

                    globe.rotation.x = TOLEDO_X;

                    targetRotationY = TOLEDO_Y;

                    targetRotationX = TOLEDO_X;

                }

                

                const elapsed = Date.now() - introStartTime;

                const progress = Math.min(elapsed / introDuration, 1);

                

                // Easing function for smooth deceleration

                const easeOutCubic = 1 - Math.pow(1 - progress, 3);

                

                // Zoom from far (8) to close (1.8)

                camera.position.z = 8 - (6.2 * easeOutCubic);

                

                if (progress >= 1) {

                    introComplete = true;

                }

            } else {

                // Handle zoom transition animation

                if (isTransitioning) {

                    const elapsed = Date.now() - transitionStartTime;

                    

                    if (transitionPhase === 'zoom-out') {

                        const progress = Math.min(elapsed / zoomOutDuration, 1);

                        const easeOut = 1 - Math.pow(1 - progress, 2);

                        

                        // Zoom out

                        const startZ = camera.position.z;

                        camera.position.z = startZ + (zoomOutDistance - startZ) * easeOut * 0.1;

                        

                        if (progress >= 1) {

                            // Switch to rotate phase - apply pending rotation

                            targetRotationY = pendingRotationY;

                            targetRotationX = pendingRotationX;

                            transitionPhase = 'zoom-in';

                            transitionStartTime = Date.now();

                        }

                    } else if (transitionPhase === 'zoom-in') {

                        const progress = Math.min(elapsed / zoomInDuration, 1);

                        const easeOut = 1 - Math.pow(1 - progress, 3);

                        

                        // Zoom in

                        camera.position.z = zoomOutDistance - (zoomOutDistance - zoomInDistance) * easeOut;

                        

                        // Check if globe rotation has settled (close enough to target)

                        const rotationSettled = 

                            Math.abs(globe.rotation.y - targetRotationY) < 0.05 &&

                            Math.abs(globe.rotation.x - targetRotationX) < 0.05;

                        

                        if (progress >= 1 && rotationSettled) {

                            isTransitioning = false;

                            transitionPhase = 'none';

                            camera.position.z = zoomInDistance;

                            

                            // Trigger bounce animation for the target location

                            if (pendingLocation) {

                                triggerBounce(pendingLocation.id);

                                // Panel no longer auto-opens - user must click icon

                                pendingLocation = null;

                            }

                        }

                    }

                }

                

                // Normal interaction after intro

                

                // Only auto-rotate if:

                // - User is not dragging

                // - Panel is not open (user is not reading)

                // - Not transitioning

                if (!isUserInteracting && !panelIsOpen && !isTransitioning) {

                    targetRotationY += 0.0001;

                }

                

                // Always smooth-lerp toward target rotation

                // This handles: auto-rotate, user drag, and clicking to new location

                globe.rotation.y += (targetRotationY - globe.rotation.y) * 0.05;

                globe.rotation.x += (targetRotationX - globe.rotation.x) * 0.05;

            }

            

            // Marker animations

            markers.forEach((marker, i) => {

                const isDoor = marker.userData.isDoor;

                const baseScale = marker.userData.baseSize * 2; // Original scale

                const isBouncing = bouncingMarkerId && marker.userData.id === bouncingMarkerId;

                

                if (isBouncing && bounceStartTime) {

                    // Bounce animation - up and down with decreasing amplitude

                    const elapsed = Date.now() - bounceStartTime;

                    const progress = elapsed / bounceDuration;

                    

                    if (progress < 1) {

                        // Bouncing: scale pulses with decreasing amplitude

                        const bounceFreq = 8; // Number of bounces

                        const decay = 1 - progress; // Decreasing amplitude

                        const bounce = 1 + Math.sin(progress * Math.PI * bounceFreq) * 0.4 * decay;

                        marker.scale.set(baseScale * bounce, baseScale * bounce, 1);

                    } else {

                        // Animation complete

                        bouncingMarkerId = null;

                        bounceStartTime = null;

                        marker.scale.set(baseScale, baseScale, 1);

                    }

                } else if (isDoor && !isBouncing) {

                    // Subtle pulse effect for doors

                    const pulse = 1 + Math.sin(Date.now() * 0.003 + i) * 0.1;

                    if (!hoveredMarker || hoveredMarker !== marker) {

                        marker.scale.set(baseScale * pulse, baseScale * pulse, 1);

                    }

                }

            });

            

            // Render Door #1 3D animation

            renderDoor1Frame();

            

            // Render Door #2 3D animation

            renderDoor2Frame();

            

            renderer.render(scene, camera);

        }

        function initStars() {

            const canvas = document.getElementById('stars');

            const ctx = canvas.getContext('2d');

            canvas.width = window.innerWidth;

            canvas.height = window.innerHeight;

            

            const stars = [];

            for (let i = 0; i < 300; i++) {

                stars.push({

                    x: Math.random() * canvas.width,

                    y: Math.random() * canvas.height,

                    size: Math.random() * 1.5 + 0.5,

                    opacity: Math.random() * 0.5 + 0.3,

                    twinkle: Math.random() * Math.PI

                });

            }

            

            function drawStars() {

                ctx.clearRect(0, 0, canvas.width, canvas.height);

                stars.forEach(star => {

                    const opacity = star.opacity + Math.sin(Date.now() * 0.001 + star.twinkle) * 0.2;

                    ctx.beginPath();

                    ctx.arc(star.x, star.y, star.size, 0, Math.PI * 2);

                    ctx.fillStyle = `rgba(245, 240, 232, ${Math.max(0.1, opacity)})`;

                    ctx.fill();

                });

                requestAnimationFrame(drawStars);

            }

            

            drawStars();

            

            window.addEventListener('resize', () => {

                canvas.width = window.innerWidth;

                canvas.height = window.innerHeight;

            });

        }
